SELECT a.id, a.region, a.mechanism_name, a.remarks, b.create_by,b.create_time, b.error_remarks, b.statues  FROM inspection_water a LEFT JOIN  (SELECT * FROM (SELECT * FROM inspection_water_record ORDER BY id DESC LIMIT 1000) t GROUP BY t.water_name_id ) b ON  a.id = b.water_name_id  order by statues desc , b.create_time desc

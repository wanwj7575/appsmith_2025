SELECT po.problem_id,po.name,po.status,po.category,po.business,po.create_by,po.create_time,po.done_time,po.deadline_time,po.accepted_by,pr.follow_record  FROM problem_order po  
left join (SELECT pr .*
FROM problem_record pr 
INNER JOIN (
    SELECT problem_id, MAX(create_time ) AS max_time
    FROM problem_record 
    GROUP BY problem_id
) pr2 ON pr.problem_id  = pr2.problem_id  AND pr.create_time = pr2.max_time) pr on po.problem_id = pr.problem_id

(po.status = {{Select1Copy.selectedOptionValue? Select1Copy.selectedOptionValue:0}} or 
{{Select1Copy.selectedOptionValue? 0:1}}) and 
(po.deadline_time < {{Checkbox1.isChecked? moment().format("YYYY-MM-DD HH:mm:ss"):0}} or {{Checkbox1.isChecked? 0:1}}) and 
po.status != 1 and po.status !=2  order by po.create_time desc ;

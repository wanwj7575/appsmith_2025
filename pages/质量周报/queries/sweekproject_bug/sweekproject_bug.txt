SELECT  zp.name ,zt.name as task_name, COUNT(1) as count_bug  FROM  zt_project zp 
left join  zt_task zt on zp.id = zt.project 
left join  (select * from zt_bug  zb where  zb .deleted =1 and resolution != 'notrepro' and  resolution != 'bydesign' and  resolution != 'willnotfix')zb on zp.id = zb.project 
WHERE  zt.closedDate >= '2025-07-26' and zt.closedDate <= '2025-08-03' and zt.`type` = 'test_task' and zp.id != 1130 group by  zp.id

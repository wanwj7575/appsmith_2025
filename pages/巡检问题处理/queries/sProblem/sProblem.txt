select  ir.inspection_record_id ,ir.problem_name ,ic.business,ir.business as new_business  ,ir.remarks ,ir.problem_type ,ir.create_time ,ir.create_by,ir.confirm,ir.error_description,tii.images from  inspection_record ir 
left join inspection_case ic ON  ir.inspectioncase_id = ic.inspectioncase_id  
left join t_inspection_images  tii on ir.inspection_record_id = tii.inspection_record_id  where ir.inspec_status = 0 and ir.confirm = 0 group by ir.inspection_record_id

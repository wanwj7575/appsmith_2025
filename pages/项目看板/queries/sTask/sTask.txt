SELECT  zt.*, za2 .comment FROM (select MAX(za.id) id  ,za.objectID,MAX(za.date) date  from zt_action za WHERE  za.objectType = 'task ' and za.comment!= '' group by za.objectID) as zal
INNER  join  zt_action za2 on zal.objectID = za2.objectID and zal.date=za2 .date
INNER join (select * from  zt_task zt WHERE id in ({{appsmith.store.select_project}})) zt on zal.objectID = zt.id
